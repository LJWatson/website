(function () {
  var btn = document.getElementById('toggle-previous-years');
  if (!btn) return;

  var targetYears = ['2013', '2014', '2015', '2016', '2017', '2018', '2019', '2021', '2022', '2023'];

  function getTargetElements() {
    var result = [];
    var h3s = document.querySelectorAll('h3');
    h3s.forEach(function (h3) {
      if (targetYears.indexOf(h3.textContent.trim()) !== -1) {
        result.push(h3);
        var el = h3.nextElementSibling;
        while (el && el.tagName !== 'H3' && el.tagName !== 'H2') {
          result.push(el);
          el = el.nextElementSibling;
        }
      }
    });
    return result;
  }

  function hide() {
    getTargetElements().forEach(function (el) { el.setAttribute('hidden', ''); });
    btn.setAttribute('aria-pressed', 'false');
  }

  function show() {
    getTargetElements().forEach(function (el) { el.removeAttribute('hidden'); });
    btn.setAttribute('aria-pressed', 'true');
  }

  hide();

  btn.addEventListener('click', function () {
    if (btn.getAttribute('aria-pressed') === 'true') {
      hide();
    } else {
      show();
    }
  });
}());
</script>
